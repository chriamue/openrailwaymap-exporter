9/// Returns the closest point of points in a given `LineString` to a given `Coord`.
10///
11/// # Examples
12///
13/// ```
14/// use geo::{line_string, coord};
15/// use openrailwaymap_exporter::algorithms::closest_point_in_linestring;
16///
17/// let linestring = line_string![
18///     coord! { x: 0.0, y: 0.0 },
19///     coord! { x: 0.0, y: 10.0 },
20///     coord! { x: 50.0, y: 50.0 },
21///     coord! { x: 100.0, y: 100.0 },
22/// ];
23/// let position = coord! { x: 10.0, y: 20.0 };
24/// let closest_position = closest_point_in_linestring(position, &linestring);
25///
26/// assert_eq!(closest_position, coord! { x: 0.0, y: 10.0 });
27/// ```
28pub fn closest_point_in_linestring(
29    position: Coord<f64>,
30    linestring: &LineString<f64>,
31) -> Coord<f64> {
32    let mut min_distance = f64::MAX;
33    let mut closest_position = position;
34
35    for p in linestring.points() {
36        let p = coord! { x: p.x(), y: p.y() };
37        let distance = p.distance(&position);
38
39        if distance.get::<meter>() < min_distance {
40            min_distance = distance.get::<meter>();
41            closest_position = p;
42        }
43    }
44
45    closest_position
46}

48/// Returns the points in front of the current location in a given `LineString`.
49///
50/// # Examples
51///
52/// ```
53/// use geo::{line_string, coord};
54/// use openrailwaymap_exporter::algorithms::{closest_point_in_linestring, points_in_front};
55///
56/// let linestring = line_string![
57///     coord! { x: 0.0, y: 0.0 },
58///     coord! { x: 0.0, y: 10.0 },
59///     coord! { x: 50.0, y: 50.0 },
60///     coord! { x: 100.0, y: 100.0 },
61/// ];
62/// let current_location = coord! { x: 10.0, y: 20.0 };
63/// let target_direction = coord! { x: 100.0, y: 100.0 };
64///
65/// let points = points_in_front(&linestring, current_location, target_direction);
66///
67/// assert_eq!(points, vec![coord! { x: 50.0, y: 50.0 }, coord! { x: 100.0, y: 100.0 }]);
68/// ```
69pub fn points_in_front(
70    linestring: &LineString<f64>,
71    current_location: Coord<f64>,
72    target_direction: Coord<f64>,
73) -> Vec<Coord<f64>> {
74    let current_location_point = Point::new(current_location.x, current_location.y);
75    let target_direction_point = Point::new(target_direction.x, target_direction.y);
76
77    let first_point = linestring.into_iter().next().unwrap();
78    let last_point = linestring.into_iter().next_back().unwrap();
79
80    let distance_to_first_point = current_location.distance(first_point);
81    let distance_to_last_point = target_direction.distance(last_point);
82
83    let linestring = if distance_to_first_point < distance_to_last_point {
84        LineString::from(
85            linestring
86                .into_iter()
87                .rev()
88                .cloned()
89                .collect::<Vec<Coord>>(),
90        )
91    } else {
92        linestring.clone()
93    };
94
95    let target_vector = target_direction_point - current_location_point;
96    let mut points_in_front = Vec::new();
97
98    for p in linestring.points() {
99        let point_vector = p - current_location_point;
100
101        if target_vector.dot(point_vector) >= 0.0 {
102            points_in_front.push(coord! { x: p.x(), y: p.y() });
103        }
104    }
105
106    points_in_front
107}

109/// Determines if the middle coordinate is between the start and end coordinates along both x and y axes.
110///
111/// This function assumes the three coordinates are collinear.
112///
113/// # Arguments
114///
115/// * `start_coord` - A `Coord<f64>` representing the start coordinate.
116/// * `middle_coord` - A `Coord<f64>` representing the middle coordinate.
117/// * `end_coord` - A `Coord<f64>` representing the end coordinate.
118///
119/// # Returns
120///
121/// A `bool` indicating whether `middle_coord` is between `start_coord` and `end_coord`.
122///
123pub fn is_middle_coord_between(
124    start_coord: Coord<f64>,
125    middle_coord: Coord<f64>,
126    end_coord: Coord<f64>,
127) -> bool {
128    let distance_start_end = start_coord.distance(&end_coord);
129    let distance_start_middle = start_coord.distance(&middle_coord);
130    let distance_middle_end = middle_coord.distance(&end_coord);
131
132    distance_start_end > distance_start_middle && distance_start_end > distance_middle_end
133}

141    fn test_closest_point_in_linestring() {
142        let linestring = line_string![
143            coord! { x: 0.0, y: 0.0 },

160        let closest_position = closest_point_in_linestring(current_location, &linestring);
