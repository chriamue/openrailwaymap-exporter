1//! Shared HTTP client construction for railway API clients.
2use reqwest::Client;
3
4/// Creates a `reqwest::Client` for railway API requests.
5///
6/// Overpass rejects requests without a User-Agent with 406 Not Acceptable.
7/// Browsers set their own User-Agent, so it is only set on native targets.
8pub fn http_client() -> Client {
9    Client::builder()
10        .user_agent(concat!(
11            env!("CARGO_PKG_NAME"),
12            "/",
13            env!("CARGO_PKG_VERSION")
14        ))
15        .build()
16        .unwrap_or_default()
17}

27        let expected_user_agent =
28            format!("{}/{}", env!("CARGO_PKG_NAME"), env!("CARGO_PKG_VERSION"));
29
30        let mut server = Server::new_async().await;
31        let mock = server
32            .mock("GET", "/")
33            .match_header("user-agent", expected_user_agent.as_str())
34            .with_status(200)
35            .create_async()
36            .await;
37
38        let client = http_client();
39        let result = client.get(server.url()).send().await;
40
41        mock.assert_async().await;
42        assert!(result.is_ok());
43    }
