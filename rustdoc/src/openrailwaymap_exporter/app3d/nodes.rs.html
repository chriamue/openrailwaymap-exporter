12/// Represents a node in the railway graph.

14pub struct Node {
15    pub id: NodeId,
16}

18/// Keeps track of the currently selected start and end nodes.
19#[derive(Default, Resource)]
20pub struct SelectedNode {
21    pub start_node_id: Option<NodeId>,
22    pub end_node_id: Option<NodeId>,
23}

25#[derive(Debug, Message)]
26pub struct NodeSelectedEvent(pub Entity);
27
28#[allow(clippy::too_many_arguments)]
29pub fn select_node_system(
30    mut events: MessageReader<NodeSelectedEvent>,
31    app_resource: Res<AppResource>,
32    mut selected_node: ResMut<SelectedNode>,
33    q_node: Query<(Entity, &Node, &Transform), Without<Camera>>,
34    interaction_mode: Res<InteractionModeResource>,
35    mut commands: Commands,
36    materials: ResMut<Assets<StandardMaterial>>,
37    asset_server: Res<AssetServer>,
38) {
39    let mut selection = None;
40    for select_event in events.read() {
41        if let Ok((entity, node, transform)) = q_node.get(select_event.0) {
42            selection = Some((entity, node.id, *transform));
43        }
44    }
45
46    if let Some((entity, node_id, transform)) = selection {
47        // Check the current interaction mode
48        match interaction_mode.mode {
49            InteractionMode::SelectMode => {
50                selected_node.end_node_id = selected_node.start_node_id;
51                selected_node.start_node_id = Some(node_id);
52                if let Some(graph) = &app_resource.graph {
53                    let edges = graph.get_edges_of_node(node_id);
54                    println!(
55                        "Selected node: {:?} {:?} {:?}",
56                        entity,
57                        graph.get_node_by_id(node_id),
58                        edges
59                            .iter()
60                            .map(|e| (e.id, e.source, e.target))
61                            .collect::<Vec<_>>()
62                    );
63                }
64            }
65            InteractionMode::PlaceTrain => {
66                if let Some(simulation) = &app_resource.simulation {
67                    let mut sim = simulation.write().unwrap();
68                    let id = create_new_train_id();
69                    let id = create_train(id, Some(node_id), None, &mut sim);
70                    let train_agent = TrainAgent::new(id);
71
72                    println!("Placing train on node: {:?}", node_id);
73                    commands
74                        .spawn((
75                            Transform::from_xyz(
76                                transform.translation.x,
77                                transform.translation.y,
78                                transform.translation.z + 1.0,
79                            ),
80                            GlobalTransform::default(),
81                            InheritedVisibility::default(),
82                            Visibility::Inherited,
83                            train_agent,
84                        ))
85                        .insert(Pickable::default())
86                        .with_children(train_agent::create_train_agent_bundle(
87                            materials,
88                            asset_server,
89                        ));
90                }
91            }
92        }
93    }
94}
