1//! The `ai` module provides a reinforcement learning agent that controls a train in a railway network simulation.
2//!
3//! The module defines the `TrainAgentState` struct, which represents the state of a train agent in the simulation,
4//! and the `TrainAgentAction` enum, which represents the possible actions a train agent can take in the simulation.
5//!
6//! The main component of this module is the `TrainAgentRL` struct, which implements the `rurel::mdp::Agent` trait
7//! for the train agent. This struct handles the reinforcement learning process, updating the train agent's state
8//! based on the actions taken and the rewards received.
9//!

31/// A reinforcement learning agent that controls a train in the simulation.
32#[derive(Default, Clone)]
33pub struct TrainAgentAI {
34    /// The id of the railway object
35    pub id: RailwayObjectId,
36    /// The railway graph representing the train network.
37    pub railway_graph: Option<RailwayGraph>,
38    /// The current node
39    pub current_node: Option<NodeId>,
40    /// The target node
41    pub target_node: Option<NodeId>,
42    /// The reinforcement learning agent responsible for controlling the train.
43    pub agent_rl: TrainAgentRL,
44    /// The trainer responsible for training the reinforcement learning agent.
45    pub trainer: Arc<RwLock<AgentTrainer<TrainAgentState>>>,
46}
47
48impl fmt::Debug for TrainAgentAI {
49    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
50        f.debug_struct("TrainAgentAI")
51            .field("railway_graph", &self.railway_graph)
52            .field("agent_rl", &self.agent_rl)
53            // We use `Arc::as_ptr()` to display the pointer value of the `AgentTrainer` to avoid
54            // issues with its non-Debug fields.
55            .field("trainer", &format_args!("{:p}", Arc::as_ptr(&self.trainer)))
56            .finish()
57    }
58}
59
60impl TrainAgentAI {
61    /// Creates a new `TrainAgentAI` with the given railway graph and initial state.
62    ///
63    /// # Arguments
64    ///
65    /// * `railway_graph` - The railway graph representing the train network.
66    /// * `initial_state` - The initial state of the train agent in the simulation.
67    ///
68    /// # Returns
69    ///
70    /// A new `TrainAgentAI` instance.
71    pub fn new(railway_graph: RailwayGraph, initial_state: TrainAgentState) -> Self {
72        let agent_rl = TrainAgentRL {
73            state: initial_state,
74            max_speed_mm_s: ((160.0 / 3.6) as i32) * 1000,
75        };
76        let trainer = Arc::new(RwLock::new(AgentTrainer::new()));
77        Self {
78            id: 0,
79            railway_graph: Some(railway_graph),
80            current_node: None,
81            target_node: None,
82            agent_rl,
83            trainer,
84        }
85    }
86
87    /// Trains the reinforcement learning agent for the specified number of iterations.
88    ///
89    /// # Arguments
90    ///
91    /// * `iterations` - The number of iterations to train the agent.
92    pub fn train(&mut self, iterations: usize) {
93        println!("Starting training for {} iterations...", iterations);
94        let mut agent = self.agent_rl.clone();
95        let mut trainer = self.trainer.write().unwrap();
96        trainer.train(
97            &mut agent,
98            &QLearning::new(0.2, 0.01, 20.),
99            &mut FixedIterations::new(iterations as u32),
100            &RandomExploration::new(),
101        );
102    }
103
104    /// Returns the best action for the given state according to the trained reinforcement learning agent.
105    ///
106    /// # Arguments
107    ///
108    /// * `state` - The current state of the train agent in the simulation.
109    ///
110    /// # Returns
111    ///
112    /// The best action for the given state or `None` if no action can be selected.
113    pub fn best_action(&self, state: &TrainAgentState) -> Option<RailMovableAction> {
114        Some(
115            self.trainer
116                .read()
117                .unwrap()
118                .best_action(state)
119                .unwrap_or_default(),
120        )
121    }
122
123    /// Updates the current edge, target node (optionally), and calculates the new state by
124    /// updating its distance using the shortest path distance while keeping the current speed constant.
125    ///
126    /// # Arguments
127    ///
128    /// * `current_edge` - The current edge on which the train agent is.
129    /// * `target_node` - The optional target node the train agent is heading to.
130    /// * `time_delta_ms` - The time delta in milliseconds since the last update.
131    pub fn observe(
132        &mut self,
133        current_node: NodeId,
134        target_node: Option<NodeId>,
135        speed_mm_s: Option<i32>,
136        delta_distance_mm: Option<i32>,
137    ) {
138        self.current_node = Some(current_node);
139        self.target_node = target_node;
140
141        let mut agent_state = self.agent_rl.state.clone();
142
143        if let Some(speed) = speed_mm_s {
144            agent_state.current_speed_mm_s = speed;
145            agent_state.max_speed_percentage = 100 * speed / self.agent_rl.max_speed_mm_s;
146        }
147
148        if let Some(delta_distance_mm) = delta_distance_mm {
149            agent_state.delta_distance_mm = delta_distance_mm;
150        }
151        self.agent_rl.state = agent_state;
152    }
153}
154
155impl DecisionAgent for TrainAgentAI {
156    type A = RailMovableAction;
157
158    fn next_action(&self, _delta_time: Option<std::time::Duration>) -> Self::A {
159        self.best_action(&self.agent_rl.state).unwrap_or_default()
160    }
161
162    fn observe(&mut self, environment: &SimulationEnvironment) {
163        if let Some(object) = environment.get_objects().iter().find(|o| o.id() == self.id) {
164            self.current_node = object.position();
165            self.target_node = object.next_target();
166            let mut agent_state = self.agent_rl.state.clone();
167
168            let speed = object.speed();
169
170            agent_state.current_speed_mm_s = speed.get::<millimeter_per_second>() as i32;
171            agent_state.max_speed_percentage = (100.0 * speed.get::<millimeter_per_second>()
172                / self.agent_rl.max_speed_mm_s as f64)
173                as i32;
174
175            self.agent_rl.state = agent_state;
176        }
177    }
178
179    fn as_any(&self) -> &dyn Any {
180        self
181    }
182}

184#[cfg(test)]
185mod tests {

187    use crate::tests::test_graph_vilbel;
188    #[test]
189    fn test_train_agent_ai() {
190        let graph = test_graph_vilbel();
191
192        let mut train_agent_ai = TrainAgentAI::new(graph, Default::default());
193
194        let source_node = 662529467;
195        let target_node = 662529466;
196
197        train_agent_ai.observe(source_node, Some(target_node), Some(1000), Some(1000));
198        train_agent_ai.train(10000);
199
200        let state = TrainAgentState {
201            delta_distance_mm: 1000,
202            current_speed_mm_s: 1000,
203            max_speed_percentage: 20,
204        };
205
206        let action = train_agent_ai.best_action(&state);
207        assert_ne!(action, None);
208    }
