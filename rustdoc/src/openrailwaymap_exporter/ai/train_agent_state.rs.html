1use crate::simulation::agents::RailMovableAction;
2use rand::RngExt;
3use rurel::mdp::State;

5/// Represents the state of a train agent in the simulation.
6#[derive(PartialEq, Eq, Hash, Clone, Debug, Default)]
7pub struct TrainAgentState {
8    /// The remaining distance in millimeters the train agent needs to travel.
9    pub delta_distance_mm: i32,
10    /// The current speed of the train agent in millimeters per second (mm/s).
11    pub current_speed_mm_s: i32,
12    /// The maximum speed percentage the train agent can reach (e.g., 100 for 100% of the maximum speed).
13    pub max_speed_percentage: i32,
14}
15
16impl TrainAgentState {
17    const MAX_ACCELERATION: i32 = 1000; // 1000 mm/s², approximately 1 m/s²
18    const ACCELERATION_STEP: i32 = 20;
19
20    fn speed_reward(&self) -> f64 {
21        (self.max_speed_percentage as f64 / 100.0).powi(2)
22    }
23
24    fn distance_reward(&self) -> f64 {
25        self.delta_distance_mm as f64
26    }
27}
28
29impl State for TrainAgentState {
30    type A = RailMovableAction;
31
32    fn reward(&self) -> f64 {
33        20.0 * self.speed_reward() + self.distance_reward()
34    }
35
36    fn actions(&self) -> Vec<Self::A> {
37        let mut actions = vec![Self::A::Stop];
38        for acceleration in 1..=(Self::MAX_ACCELERATION / Self::ACCELERATION_STEP) {
39            actions.push(Self::A::AccelerateForward {
40                acceleration: acceleration * Self::ACCELERATION_STEP,
41            });
42            actions.push(Self::A::AccelerateBackward {
43                acceleration: acceleration * Self::ACCELERATION_STEP,
44            });
45        }
46        actions
47    }
48
49    fn random_action(&self) -> Self::A {
50        let actions = self.actions();
51        let a_t = rand::rng().random_range(0..actions.len());
52        actions[a_t].clone()
53    }
54}
