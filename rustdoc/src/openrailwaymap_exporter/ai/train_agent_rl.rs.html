1use crate::ai::TrainAgentState;
2use crate::simulation::agents::RailMovableAction;
3use rurel::mdp::{Agent, State};
4
5/// Reinforcement Learning Agent for controlling a train in the simulation.
6#[derive(Default, Clone, Debug)]
7pub struct TrainAgentRL {
8    /// The current state of the train agent.
9    pub state: TrainAgentState,
10    /// The maximum speed the train agent can reach in millimeters per second (mm/s).
11    pub max_speed_mm_s: i32,
12}
13
14impl TrainAgentRL {
15    const TIME_DELTA_MS: u32 = 1000;
16}
17
18impl Agent<TrainAgentState> for TrainAgentRL {
19    fn current_state(&self) -> &TrainAgentState {
20        &self.state
21    }
22
23    fn take_action(&mut self, action: &RailMovableAction) {
24        match action {
25            RailMovableAction::Stop => {
26                self.state.current_speed_mm_s = 0;
27            }
28            RailMovableAction::AccelerateForward { acceleration } => {
29                self.state.current_speed_mm_s += acceleration * Self::TIME_DELTA_MS as i32 / 1000;
30                self.state.delta_distance_mm =
31                    self.state.current_speed_mm_s * Self::TIME_DELTA_MS as i32 / 1000;
32            }
33            RailMovableAction::AccelerateBackward { acceleration } => {
34                self.state.current_speed_mm_s -= acceleration * Self::TIME_DELTA_MS as i32 / 1000;
35                self.state.delta_distance_mm =
36                    self.state.current_speed_mm_s * Self::TIME_DELTA_MS as i32 / 1000;
37            }
38        }
39        self.state.max_speed_percentage =
40            (((self.state.current_speed_mm_s as f64 / self.max_speed_mm_s as f64) * 100.0) as i32)
41                .abs();
42    }
43
44    fn pick_random_action(&mut self) -> <TrainAgentState as State>::A {
45        let action = self.current_state().random_action();
46        self.take_action(&action);
47        action
48    }
49}

51#[cfg(test)]
52mod tests {
53    use super::*;
54
55    #[test]
56    fn test_take_action() {
57        let mut agent = TrainAgentRL {
58            state: TrainAgentState {
59                delta_distance_mm: 1000,
60                current_speed_mm_s: 0,
61                max_speed_percentage: 0,
62            },
63            max_speed_mm_s: (1000.0 * 160.0 / 3.6) as i32,
64        };
65
66        // Test AccelerateForward action
67        agent.take_action(&RailMovableAction::AccelerateForward { acceleration: 1000 });
68        assert_eq!(agent.state.current_speed_mm_s, 1000);
69        assert_eq!(agent.state.delta_distance_mm, 1000);
70
71        // Test AccelerateForward action
72        agent.take_action(&RailMovableAction::AccelerateForward { acceleration: 500 });
73        assert_eq!(agent.state.current_speed_mm_s, 1500);
74        assert_eq!(agent.state.delta_distance_mm, 1500);
75
76        // Test AccelerateBackward action
77        agent.take_action(&RailMovableAction::AccelerateBackward { acceleration: 500 });
78        assert_eq!(agent.state.current_speed_mm_s, 1000);
79        assert_eq!(agent.state.delta_distance_mm, 1000);
80
81        // Test Stop action
82        agent.take_action(&RailMovableAction::Stop);
83        assert_eq!(agent.state.current_speed_mm_s, 0);
84    }
