1use crate::prelude::RailwayEdge;
2use yew::prelude::*;
3
4/// A Yew component for visualizing railway edges as SVG paths.
5pub struct SvgEdge {
6    hovered: bool,
7}
8
9#[derive(PartialEq, Properties, Clone)]
10pub struct Props {
11    /// The railway edge to visualize.
12    pub edge: RailwayEdge,
13    /// The scaling factor for the x-axis.
14    pub scale_x: f64,
15    /// The scaling factor for the y-axis.
16    pub scale_y: f64,
17    /// The height of the SVG view.
18    pub view_height: f64,
19    /// The minimum coordinates of the graph's bounding box.
20    pub min_coord: (f64, f64),
21    #[prop_or_default]
22    pub stroke_color: Option<String>,
23}
24
25/// Messages for the `SvgEdge` component.
26pub enum Msg {
27    MouseEnter,
28    MouseLeave,
29}
30
31impl Component for SvgEdge {
32    type Message = Msg;
33    type Properties = Props;
34
35    fn create(_ctx: &Context<Self>) -> Self {
36        SvgEdge { hovered: false }
37    }
38
39    fn update(&mut self, _ctx: &Context<Self>, msg: Self::Message) -> bool {
40        match msg {
41            Msg::MouseEnter => {
42                self.hovered = true;
43            }
44            Msg::MouseLeave => {
45                self.hovered = false;
46            }
47        }
48        true
49    }
50
51    fn view(&self, ctx: &Context<Self>) -> Html {
52        let edge_data = &ctx.props().edge;
53        let path_data: String = edge_data
54            .path
55            .0
56            .iter()
57            .enumerate()
58            .map(|(i, coord)| {
59                let x = (coord.x - ctx.props().min_coord.0) * ctx.props().scale_x;
60                let y = ctx.props().view_height
61                    - (coord.y - ctx.props().min_coord.1) * ctx.props().scale_y;
62
63                if i == 0 {
64                    format!("M {:.1} {:.1}", x, y)
65                } else {
66                    format!(" L {:.1} {:.1}", x, y)
67                }
68            })
69            .collect::<Vec<String>>()
70            .join("");
71
72        let stroke_width = if self.hovered { 2 } else { 1 };
73
74        html! {
75            <path
76                d={path_data}
77                stroke={ctx.props().stroke_color.as_ref().unwrap_or(&"black".to_string()).to_string()}
78                stroke-width={format!("{}", stroke_width)}
79                fill="none"
80                onmouseover={ctx.link().callback(|_| Msg::MouseEnter)}
81                onmouseout={ctx.link().callback(|_| Msg::MouseLeave)}
82            />
83        }
84    }
85}

96        let edge = RailwayEdge {
97            id: 1,
98            length: 1500.0,
99            path: LineString::from(vec![
100                coord! { x: 8.6821, y: 50.1109 },
101                coord! { x: 8.6921, y: 50.1209 },
102            ]),
103            source: 0,
104            target: 0,
105        };
106        let props = Props {
107            edge,
108            scale_x: 1.0,
109            scale_y: 1.0,
110            view_height: 100.0,
111            min_coord: (8.6800, 50.1000),
112            stroke_color: None,
113        };
114
115        let rendered = LocalServerRenderer::<SvgEdge>::with_props(props)
116            .render()
117            .await;

119        assert!(rendered.contains("<path"));
