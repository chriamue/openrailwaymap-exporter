1//! A module containing the `SvgComponent`, which renders a `RailwayGraph` as an SVG.

12/// Properties for the `SvgComponent`.
13#[derive(PartialEq, Properties, Clone)]
14pub struct Props {
15    /// The `RailwayGraph` to render.
16    pub graph: Option<RailwayGraph>,
17    /// The width of the SVG view.
18    pub view_width: f64,
19    /// The height of the SVG view.
20    pub view_height: f64,
21    /// A callback that gets called when a node is selected.
22    pub on_select_node: Option<Callback<NodeId>>,
23    /// The starting node ID for pathfinding.
24    pub start_node_id: Option<NodeId>,
25    /// The ending node ID for pathfinding.
26    pub end_node_id: Option<NodeId>,
27}
28
29/// A component that renders a `RailwayGraph` as an SVG.
30pub struct SvgComponent {}
31
32impl Component for SvgComponent {
33    type Message = ();
34    type Properties = Props;
35
36    fn create(_: &Context<Self>) -> Self {
37        Self {}
38    }
39
40    fn update(&mut self, _: &Context<Self>, _: Self::Message) -> bool {
41        false
42    }
43
44    fn view(&self, ctx: &Context<Self>) -> Html {
45        if let Some(graph) = ctx.props().graph.as_ref() {
46            let (min_coord, max_coord) = graph.bounding_box();
47            let scale_x = ctx.props().view_width / (max_coord.x - min_coord.x);
48            let scale_y = ctx.props().view_height / (max_coord.y - min_coord.y);
49
50            let svg_edges: Vec<Html> = graph
51                .physical_graph.graph
52                .edge_references()
53                .map(|edge| {
54                    let edge_data = edge.weight();
55
56                    html! {
57                        <SvgEdge edge={edge_data.clone()} scale_x={scale_x} scale_y={scale_y} min_coord={(min_coord.x, min_coord.y)} view_height={ctx.props().view_height} />
58                    }
59                })
60                .collect();
61
62            let svg_nodes: Vec<Html> = graph
63                .physical_graph.graph
64                .node_references()
65                .map(|node| {
66                    let node_data = node.weight();
67
68                    html! {
69                        <SvgNode node={*node_data} scale_x={scale_x} scale_y={scale_y}
70                         min_coord={(min_coord.x, min_coord.y)} view_height={ctx.props().view_height}
71                         on_select={ctx.props().on_select_node.clone()} />
72                    }
73                })
74                .collect();
75
76            let path_edges: Option<Vec<RailwayEdge>> =
77                if let (Some(start_node_id), Some(end_node_id)) =
78                    (ctx.props().start_node_id, ctx.props().end_node_id)
79                {
80                    // Use graph.shortest_path_edges to get the Vec of edge IDs
81                    let path_edge_ids = graph.shortest_path_edges(start_node_id, end_node_id);
82
83                    // If path_edge_ids is Some, map the edge IDs to RailwayEdge instances
84                    path_edge_ids.map(|ids| {
85                        ids.into_iter()
86                            .filter_map(|id| {
87                                graph
88                                    .physical_graph
89                                    .graph
90                                    .edge_references()
91                                    .find(|edge| edge.weight().id == id)
92                                    .map(|edge| edge.weight().clone())
93                            })
94                            .collect::<Vec<RailwayEdge>>()
95                    })
96                } else {
97                    None
98                };
99
100            let path_edges: Vec<Html> = path_edges
101                .unwrap_or_default()
102                .iter()
103                .map(|edge_data| {
104                    html! {
105                        <SvgEdge
106                            edge={edge_data.clone()}
107                            scale_x={scale_x}
108                            scale_y={scale_y}
109                            min_coord={(min_coord.x, min_coord.y)}
110                            view_height={ctx.props().view_height}
111                            stroke_color={Some("red".to_string())}
112                        />
113                    }
114                })
115                .collect();
116
117            html! {
118                <svg xmlns="http://www.w3.org/2000/svg" viewBox={format!("0 0 {} {}", ctx.props().view_width, ctx.props().view_height)}>
119                    { for svg_edges }
120                    { for svg_nodes }
121                    { for path_edges }
122                </svg>
123            }
124        } else {
125            html! { <></> }
126        }
127    }
128}

130#[cfg(test)]
131mod tests {
132    use std::collections::HashMap;
133
134    use super::*;
135    use crate::importer::overpass_importer::{
136        from_railway_elements, Coordinate, ElementType, RailwayElement,
137    };
138    use yew::LocalServerRenderer;
139
140    #[cfg_attr(not(target_arch = "wasm32"), tokio::test)]
141    #[cfg_attr(target_arch = "wasm32", wasm_bindgen_test::wasm_bindgen_test)]
142    async fn test_render() {
143        let elements = vec![
144            RailwayElement {
145                id: 1,
146                element_type: ElementType::Node,
147                lat: Some(50.1109),
148                lon: Some(8.6821),
149                tags: Some(HashMap::new()),
150                nodes: None,
151                geometry: None,
152            },
153            RailwayElement {
154                id: 2,
155                element_type: ElementType::Node,
156                lat: Some(50.1122),
157                lon: Some(8.6833),
158                tags: Some(HashMap::new()),
159                nodes: None,
160                geometry: None,
161            },
162            RailwayElement {
163                id: 3,
164                element_type: ElementType::Way,
165                lat: None,
166                lon: None,
167                tags: Some(HashMap::new()),
168                nodes: Some(vec![1, 2]),
169                geometry: Some(vec![
170                    Coordinate {
171                        lat: 50.1109,
172                        lon: 8.6821,
173                    },
174                    Coordinate {
175                        lat: 50.1122,
176                        lon: 8.6833,
177                    },
178                ]),
179            },
180        ];
181
182        let railway_graph = from_railway_elements(&elements);
183        let props = Props {
184            graph: Some(railway_graph),
185            view_width: 200.0,
186            view_height: 100.0,
187            on_select_node: None,
188            start_node_id: None,
189            end_node_id: None,
190        };
191
192        let rendered = LocalServerRenderer::<SvgComponent>::with_props(props)
193            .render()
194            .await;
195
196        assert!(rendered.contains("<svg"));
197        assert!(rendered.contains("<path"));
198        assert!(rendered.contains("stroke=\"black\""));
199        assert!(rendered.contains("stroke-width=\"1\""));
200        assert!(rendered.contains("<circle"));
201        assert!(rendered.contains("cx="));
202        assert!(rendered.contains("cy="));
203        assert!(rendered.contains("r=\"2\""));
204        assert!(rendered.contains("fill=\"red\""));
205    }
206}
