44/// A trait that defines an object within the simulation that can move along a railway.
45pub trait SimulationObject:
46    RailwayObject + Movable + NextTarget + GeoLocation + Send + Sync
47{
48}
49impl<T: RailwayObject + Movable + NextTarget + GeoLocation + Send + Sync> SimulationObject for T {}

51/// A `Simulation` struct holding a railway graph and a list of moveable railway objects.
52pub struct Simulation {
53    /// The simulation environment
54    pub environment: SimulationEnvironment,
55    /// A list of agents
56    pub object_agents: HashMap<RailwayObjectId, Box<dyn DecisionAgent<A = RailMovableAction>>>,
57    /// A list of metrics handlers
58    pub metrics_handlers: Vec<Box<dyn MetricsHandler>>,
59    /// Elapsed time of simulation
60    elapsed_time: Duration,
61    /// simulation pause state
62    pub is_paused: bool,
63    /// Speedup factor of the simulation
64    pub speedup_factor: f64,
65}
66
67unsafe impl Send for Simulation {}
68unsafe impl Sync for Simulation {}
69
70impl fmt::Debug for Simulation {
71    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
72        f.debug_struct("Simulation")
73            .field("railway_objects", &self.environment.objects.len())
74            .finish()
75    }
76}

79    /// Creates a new simulation with the given railway graph.
80    ///
81    /// # Arguments

83    /// * `graph` - The railway graph representing the railway infrastructure.

85    /// # Returns
86    ///
87    /// A new `Simulation` instance.

89    pub fn new(graph: RailwayGraph) -> Self {
90        let default_metrics_handler: Vec<Box<dyn MetricsHandler>> = vec![
91            Box::new(ActionCountHandler::new()),
92            Box::new(TargetReachedHandler::new()),
93        ];
94
95        Self {
96            environment: SimulationEnvironment {

98                objects: HashMap::new(),
99            },
100            object_agents: HashMap::new(),
101            metrics_handlers: default_metrics_handler,
102            elapsed_time: Duration::default(),
103            is_paused: false,
104            speedup_factor: 1.0,
105        }
106    }

108    /// Returns a reference to the observable environment of the simulation.
109    ///
110    /// The observable environment allows external components to access the
111    /// state of the simulation without being able to modify it. This is useful
112    /// for agents to observe the simulation state and make decisions based on it.
113    ///
114    /// # Returns
115    ///
116    /// A reference to a trait object implementing the `ObservableEnvironment` trait,
117    /// which provides read-only access to the simulation environment.
118    ///
119    pub fn get_observable_environment(&self) -> &(dyn ObservableEnvironment + 'static) {
120        self.environment.as_observable_env()
121    }

123    /// Adds a moveable railway object to the simulation.
124    ///
125    /// # Arguments
126    ///
127    /// * `object` - The moveable railway object to be added to the simulation.
128    ///
129    /// # Returns
130    ///
131    /// A boolean indicating if the object was successfully added.
132    ///
133    pub fn add_object(
134        &mut self,
135        object: Box<dyn SimulationObject>,
136        agent: Option<Box<dyn DecisionAgent<A = RailMovableAction>>>,
137    ) -> bool {
138        if let std::collections::hash_map::Entry::Vacant(e) =
139            self.environment.objects.entry(object.id())
140        {
141            let id = object.id();
142            e.insert(object);
143
144            if let Some(agent) = agent {
145                self.add_agent_for_object(id, agent);
146            }
147            true
148        } else {
149            false
150        }
151    }

154    ///
155    /// # Arguments
156    ///
157    /// * `id` - The unique identifier of the moveable railway object to be removed.

161    /// A boolean indicating if the object was successfully removed.

163    pub fn remove_object(&mut self, id: i64) -> bool {

178    pub fn add_agent_for_object(
179        &mut self,
180        object_id: RailwayObjectId,
181        agent: Box<dyn DecisionAgent<A = RailMovableAction>>,
182    ) -> bool {
183        if self.environment.objects.contains_key(&object_id) {
184            self.object_agents.insert(object_id, agent);
185            true
186        } else {
187            false
188        }
189    }

191    /// Registers a metrics handler for the simulation.
192    ///
193    /// This function adds a new metrics handler to the simulation. The metrics handler
194    /// will be used to process events and gather metrics during the simulation run.
195    ///
196    /// # Arguments
197    ///
198    /// * `handler` - A boxed metrics handler that implements the `MetricsHandler` trait.
199    ///
200    pub fn register_metrics_handler(&mut self, handler: Box<dyn MetricsHandler>) {
201        self.metrics_handlers.push(handler);
202    }

204    /// Handles a simulation event by passing it to all registered metrics handlers.
205    ///
206    /// This function is called internally by the simulation engine whenever a simulation
207    /// event occurs. It iterates through all registered metrics handlers and calls their
208    /// `handle` function with the event as an argument.
209    ///
210    /// # Arguments
211    ///
212    /// * `event` - A reference to a simulation event that implements the `SimulationEvent` trait.
213    ///
214    /// This function is not meant to be called directly by the user. It is called internally
215    /// by the simulation engine.
216    fn handle_event(&mut self, event: &dyn SimulationEvent) {
217        for handler in &mut self.metrics_handlers {
218            handler.handle(event);
219        }
220    }

222    /// Updates the simulation state based on the given delta time and the speedup factor.
223    ///
224    /// # Arguments
225    ///
226    /// * `delta_time` - The elapsed time since the last update.
227    pub fn update(&mut self, delta_time: Duration) {
228        if !self.is_paused {
229            let int_speedup_factor = self.speedup_factor.floor() as u32;
230            let fractional_speedup_factor = self.speedup_factor.fract();
231
232            let full_delta_time = delta_time;
233            let fractional_delta_time = delta_time.mul_f64(fractional_speedup_factor);
234
235            // Update the total elapsed time.
236            let scaled_delta_time = delta_time.mul_f64(self.speedup_factor);
237            self.elapsed_time += scaled_delta_time;
238
239            for _ in 0..int_speedup_factor {
240                let object_ids: Vec<_> = self.environment.objects.keys().cloned().collect();
241
242                // Iterate over each object in the simulation and update its state based on the delta time and speedup factor.
243                for id in object_ids {
244                    self.update_object(full_delta_time, id);
245                }
246            }
247
248            if fractional_speedup_factor > 0.0 {
249                for _ in 0..int_speedup_factor {
250                    let object_ids: Vec<_> = self.environment.objects.keys().cloned().collect();
251
252                    for id in object_ids {
253                        self.update_object(fractional_delta_time, id);
254                    }
255                }
256            }
257        }
258    }

260    /// Updates the state of the object with the given id based on the given delta time.
261    ///
262    /// # Arguments
263    ///
264    /// * `delta_time` - The elapsed time since the last update.
265    /// * `id` - The unique identifier of the moveable railway object to be updated.
266    fn update_object(&mut self, delta_time: Duration, id: RailwayObjectId) {
267        if let Some(agent) = self.object_agents.get_mut(&id) {
268            // Observe the environment.
269            agent.observe(&self.environment);
270        }
271        let mut event = None;
272        if let Some(object) = self.environment.objects.get_mut(&id) {
273            // Get the action from the decision agent.
274            if let Some(agent) = self.object_agents.get(&id) {
275                let action = agent.next_action(Some(delta_time));
276                // Update the acceleration based on the action.
277                match action {
278                    RailMovableAction::Stop => {
279                        let speed = object.speed();
280                        if speed.is_sign_positive() {
281                            object.set_acceleration(-object.acceleration().abs());
282                        } else if speed.is_sign_negative() {
283                            object.set_acceleration(object.acceleration().abs());
284                        } else {
285                            object.set_acceleration(Acceleration::new::<meter_per_second_squared>(
286                                0.0,
287                            ));
288                        }
289                    }
290                    RailMovableAction::AccelerateForward { acceleration } => {
291                        object.set_acceleration(Acceleration::new::<meter_per_second_squared>(
292                            acceleration as f64,
293                        ));
294                    }
295                    RailMovableAction::AccelerateBackward { acceleration } => {
296                        object.set_acceleration(Acceleration::new::<meter_per_second_squared>(
297                            -acceleration as f64,
298                        ));
299                    }
300                }
301
302                event = Some(RailMovableEvent { action });
303
304                // Update speed based on the acceleration
305                object.set_speed(object.max_speed().min(
306                    object.speed()
307                        + Time::new::<second>(delta_time.as_secs_f64()) * object.acceleration(),
308                ));
309            }
310        }
311        if let Some(event) = event {
312            self.handle_event(&event);
313        }
314
315        self.update_object_position(id, delta_time);
316        self.update_train_target(id);
317    }

319    fn update_object_position(&mut self, id: RailwayObjectId, delta_time: Duration) {
320        const NEXT_NODE_DISTANCE_TOLERANCE: f64 = 1.0;
321        if let Some(object) = self.environment.objects.get_mut(&id) {
322            if let Some(current_position) = object.position() {
323                let current_speed = object.speed();
324                let target = object.next_target();
325                if let Some(current_location) = object.geo_location() {
326                    let graph = &self.environment.graph;
327                    let next_node =
328                        graph.get_next_node(current_position, target.unwrap_or_default());
329
330                    if let Some(next_node_id) = next_node {
331                        let edge = graph
332                            .railway_edge(current_position, next_node_id)
333                            .expect("Invalid edge");
334                        let next_node = graph.get_node_by_id(next_node_id).unwrap();
335
336                        let direction_coord = next_node.location;
337                        let distance_to_travel =
338                            current_speed * Time::new::<second>(delta_time.as_secs_f64());
339
340                        let new_geo_location = edge.position_on_edge(
341                            current_location,
342                            distance_to_travel,
343                            direction_coord,
344                        );
345
346                        let current_node = graph.get_node_by_id(current_position).unwrap();
347
348                        // reached next node
349                        if is_middle_coord_between(
350                            current_node.location,
351                            next_node.location,
352                            new_geo_location,
353                        ) || edge.distance_to_end(new_geo_location, direction_coord)
354                            < Length::new::<meter>(NEXT_NODE_DISTANCE_TOLERANCE)
355                        {
356                            object.set_position(Some(next_node_id));
357                        }
358                        object.set_geo_location(Some(new_geo_location));
359                    } else {
360                        warn!("object {} has no next node, unsetting target", object.id());
361                        object.set_next_target(None);
362                    }
363                } else {
364                    warn!("object {} has no coordinates", object.id())
365                }
366            }
367        }
368    }

370    fn update_train_target(&mut self, id: RailwayObjectId) {
371        let mut event = None;
372        if let Some(object) = self.environment.objects.get_mut(&id) {
373            if let Some(train) = object.as_any_mut().downcast_mut::<Train>() {
374                if train.next_target().is_none() || train.position() == train.next_target() {
375                    let reachable_nodes = self
376                        .environment
377                        .graph
378                        .reachable_nodes(train.position().unwrap());
379                    if !reachable_nodes.is_empty() {
380                        event = Some(TargetReachedEvent {});
381                        let mut rng = rand::rng();
382                        train.set_next_target(Some(*reachable_nodes.choose(&mut rng).unwrap()));
383                    }
384                }
385            }
386        }
387        if let Some(event) = event {
388            self.handle_event(&event);
389        }
390    }
