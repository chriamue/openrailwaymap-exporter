1//! This module contains types related to handling and processing metric commands during railway simulations.
2//!
3//! `MetricsCommand` is an enumeration of subcommands for working with metrics in the simulation.
4//! It provides the `List` and `Get` subcommands to list available metrics and get a specific metric's value.
5//!
6//! This module integrates with the `Simulation` struct and the `MetricsHandler` trait to process and display metric information.
7
8use crate::simulation::commands::SimulationCommand;
9use crate::simulation::metrics::{ActionCountHandler, MetricsHandler, TargetReachedHandler};
10use crate::simulation::Simulation;
11use clap::Parser;
12
13/// An enumeration of subcommands for working with metrics in the simulation.
14#[derive(Parser, Debug)]
15#[command(name = "metrics")]
16pub enum MetricsCommand {
17    /// Lists the names of all available metrics.
18    #[command(name = "list")]
19    List,
20
21    /// Retrieves the value of a specific metric.
22    #[command(name = "get")]
23    Get(GetMetric),
24}
25
26/// A struct representing the `Get` subcommand for retrieving a specific metric's value.
27#[derive(Parser, Debug)]
28#[command(name = "get")]
29pub struct GetMetric {
30    /// The identifier of the metric to retrieve.
31    #[clap(name = "ID")]
32    pub metric_id: String,
33}
34
35/// Implements the `SimulationCommand` trait for the `MetricsCommand` enumeration.
36impl SimulationCommand for MetricsCommand {
37    /// Executes the metric command, either listing available metrics or getting a specific metric's value.
38    ///
39    /// # Returns
40    ///
41    /// An optional `String` containing the command's output.
42    fn execute(&self, simulation: &mut Simulation) -> Option<String> {
43        match self {
44            MetricsCommand::List => {
45                let metric_names = vec!["ActionCount", "TargetReached"];
46                Some(format!("Available metrics: {:?}", metric_names))
47            }
48            MetricsCommand::Get(get_metric) => {
49                let metric_value = match get_metric.metric_id.as_str() {
50                    "ActionCount" => {
51                        let handler = simulation.metrics_handlers.iter().find_map(|handler| {
52                            handler.as_any().downcast_ref::<ActionCountHandler>()
53                        });
54                        handler.map(|h| h.get_value())
55                    }
56                    "TargetReached" => {
57                        let handler = simulation.metrics_handlers.iter().find_map(|handler| {
58                            handler.as_any().downcast_ref::<TargetReachedHandler>()
59                        });
60                        handler.map(|h| h.get_value())
61                    }
62                    _ => None,
63                };
64
65                if let Some(value) = metric_value {
66                    Some(format!("{}: {}", get_metric.metric_id, value))
67                } else {
68                    Some(format!("Metric '{}' not found", get_metric.metric_id))
69                }
70            }
71        }
72    }
73}
