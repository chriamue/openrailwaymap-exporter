11/// Generates a DOT string representation of a given RailwayGraph.
12///
13/// The DOT string can be used to visualize the graph using tools like Graphviz.
14///
15/// # Arguments
16///
17/// * `graph` - A reference to a RailwayGraph.
18///
19/// # Returns
20///
21/// A `Result` containing a DOT-formatted `String` on success, or a `Box<dyn Error>` on failure.
22///
23/// # Example
24///
25/// ```
26/// use openrailwaymap_exporter::importer::overpass_importer::RailwayElement;
27/// use openrailwaymap_exporter::prelude::generate_dot_string;
28/// use openrailwaymap_exporter::importer::overpass_importer::from_railway_elements;
29///
30/// let elements = vec![
31///     RailwayElement::new_with_id(1),
32///     RailwayElement::new_with_id(2),
33/// ];
34///
35/// let railway_graph = from_railway_elements(&elements);
36/// let dot_string = generate_dot_string(&railway_graph).unwrap();
37///
38/// println!("{}", dot_string);
39/// ```
40pub fn generate_dot_string(graph: &RailwayGraph) -> Result<String, Box<dyn Error>> {
41    let dot = Dot::with_config(&graph.physical_graph.graph, &[Config::EdgeNoLabel]);
42    Ok(format!("{:?}", dot))
43}

55    #[test]
56    fn test_generate_dot_string() {
57        let elements = vec![
58            RailwayElement {
59                id: 1,
60                element_type: ElementType::Way,
61                lat: None,
62                lon: None,
63                tags: None,
64                nodes: Some(vec![2, 3]),
65                geometry: Some(vec![
66                    Coordinate {
67                        lat: 50.0,
68                        lon: 8.0,
69                    },
70                    Coordinate {
71                        lat: 51.0,
72                        lon: 9.0,
73                    },
74                ]),
75            },
76            RailwayElement {
77                id: 2,
78                element_type: ElementType::Node,
79                lat: Some(50.0),
80                lon: Some(8.0),
81                tags: None,
82                nodes: None,
83                geometry: None,
84            },
85            RailwayElement {
86                id: 3,
87                element_type: ElementType::Node,
88                lat: Some(51.0),
89                lon: Some(9.0),
90                tags: None,
91                nodes: None,
92                geometry: None,
93            },
94        ];
95
96        let railway_graph = from_railway_elements(&elements);
97        let dot_string = generate_dot_string(&railway_graph).unwrap();
98        println!("{}", &dot_string);
99
100        assert!(dot_string.contains("graph {"));
101        assert!(dot_string
102            .contains("0 [ label = \"TransitNode { id: 2, location: COORD(8.0 50.0) }\" ]"));
103        assert!(dot_string
104            .contains("1 [ label = \"TransitNode { id: 3, location: COORD(9.0 51.0) }\" ]"));
105        assert!(dot_string.contains("0 -- 1 [ ]"));
106        assert!(dot_string.contains('}'));
107    }
